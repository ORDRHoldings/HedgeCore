# Admin Hub Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the two broken admin pages (`/admin-monitor`, `/devops`) with a unified, fully-tested 8-tab Admin Hub at `/admin` covering operations, users, tenants, roles, API keys, metrics, config, and DevOps.

**Architecture:** Single Next.js page at `/admin` routes via `?tab=` param to 8 independent tab components. Each tab fetches its own data, owns its own state, and is a self-contained file. Backend unit tests cover all 6 admin route modules. One Playwright E2E spec covers auth gate + all 8 tabs.

**Tech Stack:** Next.js 15.5 App Router, React 19, TypeScript 5.9, inline CSS variables (no Tailwind on pages), `dashboardFetch` for API calls, `useAuth` for auth. Backend: FastAPI, pytest, httpx AsyncClient, `require_superuser` dependency override pattern.

**Spec:** `docs/superpowers/specs/2026-03-15-admin-hub-design.md`

function TabLoader() {
  return (
    <div style={{ display: "flex", flex: 1, alignItems: "center", justifyContent: "center" }}>
      <span style={{ fontFamily: "var(--font-terminal-mono,'IBM Plex Mono',monospace)", fontSize: 11, color: "var(--text-tertiary)", letterSpacing: "0.1em" }}>
        LOADING…
      </span>
    </div>
  );

function AdminContent() {
  const { token, user, isAuthenticated } = useAuth();
  const searchParams = useSearchParams();
  const router = useRouter();

  const tabParam = (searchParams?.get("tab") ?? "operations") as AdminTab;
  const [activeTab, setActiveTab] = useState<AdminTab>(tabParam);

  const handleTabChange = (tab: AdminTab) => {
    setActiveTab(tab);
    router.replace(`/admin?tab=${tab}`, { scroll: false });
  };

  if (!isAuthenticated || !token || !user) return <DeniedCard />;
  // Cast user to any to access is_superuser — shape depends on auth context
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  if (!(user as any).is_superuser) return <DeniedCard />;

  return (
    <div style={{ display: "flex", flexDirection: "column", flex: 1, minHeight: 0, overflow: "hidden" }}>
      <AdminTabBar active={activeTab} onChange={handleTabChange} />
      <div style={{ flex: 1, overflowY: "auto", background: S.bgDeep }}>
        <Suspense fallback={<TabLoader />}>
          {activeTab === "operations" && <OperationsTab token={token} />}
          {activeTab === "users"      && <UsersTab      token={token} />}
          {activeTab === "tenants"    && <TenantsTab    token={token} />}
          {activeTab === "roles"      && <RolesTab      token={token} />}
          {activeTab === "apikeys"    && <ApiKeysTab    token={token} />}
          {activeTab === "metrics"    && <MetricsTab    token={token} />}
          {activeTab === "config"     && <ConfigTab     token={token} />}
          {activeTab === "devops"     && <DevOpsTab     token={token} />}
        </Suspense>
      </div>
    </div>
  );
}

export default function AdminHubPage() {
  return (
    <PageShell icon={Shield} title="Admin Hub" breadcrumb={["Platform", "Admin"]} noPadding>
      <Suspense fallback={<TabLoader />}>
        <AdminContent />
      </Suspense>
    </PageShell>
  );
}

"use client";

import { useEffect, useState, useCallback, useRef } from "react";
import { dashboardFetch } from "@/lib/api/dashboardClient";

const S = {
  fontUI:   "var(--font-terminal,'IBM Plex Sans',sans-serif)",
  fontMono: "var(--font-terminal-mono,'IBM Plex Mono',monospace)",
  bgDeep:   "var(--bg-deep)",
  bgPanel:  "var(--bg-panel)",
  bgSub:    "var(--bg-sub)",
  rim:      "var(--border-rim)",
  soft:     "var(--border-soft)",
  primary:  "var(--text-primary)",
  secondary:"var(--text-secondary)",
  tertiary: "var(--text-tertiary)",
  cyan:     "var(--accent-cyan)",
  amber:    "var(--accent-amber)",
  red:      "var(--accent-red)",
  pass:     "var(--status-pass)",
  fail:     "var(--status-fail)",
} as const;

interface HealthData {
  db_latency_ms: number;
  uptime_human: string;
  memory?: { rss_mb: number; vms_mb: number } | null;
  total_users?: number;
  active_users?: number;
  total_companies?: number;
  calc_runs?: number;
}
interface ServiceInfo {
  name: string;
  status: "running" | "degraded" | "stopped";
  uptime_human: string;
}
interface TableInfo {
  table_name: string;
  row_count: number;
  last_insert?: string | null;
}
interface EngineModule {
  name: string;
  wired: boolean;
  file_exists: boolean;
}
interface ErrorGroup {
  event_type: string;
  count: number;
  latest: string;
}
interface ActivityEvent {
  id: string;
  event_type: string;
  description: string;
  actor_email?: string | null;
  company_name?: string | null;
  hash?: string | null;
  created_at: string;
}

const STATUS_COLOR: Record<string, string> = {
  running: "var(--status-pass)",
  degraded: "var(--accent-amber)",
  stopped: "var(--accent-red)",
};

const EVENT_COLOR: Record<string, string> = {
  CALCULATE: "var(--accent-cyan)",
  APPROVE:   "var(--status-pass)",
  REJECT:    "var(--accent-red)",
  AUTH:      "var(--accent-amber)",
  SYSTEM:    "var(--accent-indigo)",
  PIPELINE:  "var(--accent-indigo)",
  USER:      "var(--accent-amber)",
  POLICY:    "var(--accent-cyan)",
};

function SectionCard({ title, children }: { title: string; children: React.ReactNode }) {
  return (
    <div style={{ background: S.bgPanel, border: `1px solid ${S.rim}`, marginBottom: 16 }}>
      <div style={{ padding: "8px 16px", borderBottom: `1px solid ${S.rim}`, background: S.bgSub }}>
        <span style={{ fontFamily: S.fontMono, fontSize: 11, fontWeight: 700, letterSpacing: "0.09em", color: S.tertiary }}>
          {title}
        </span>
      </div>
      <div style={{ padding: 16 }}>{children}</div>
    </div>
  );
}

export default function OperationsTab({ token }: { token: string }) {
  const [health, setHealth] = useState<HealthData | null>(null);
  const [services, setServices] = useState<ServiceInfo[]>([]);
  const [tables, setTables] = useState<TableInfo[]>([]);
  const [engine, setEngine] = useState<EngineModule[]>([]);
  const [errors, setErrors] = useState<ErrorGroup[]>([]);
  const [activity, setActivity] = useState<ActivityEvent[]>([]);
  const [loading, setLoading] = useState(true);
  const [restarting, setRestarting] = useState<string | null>(null);
  const [toast, setToast] = useState<string | null>(null);
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  const fetchAll = useCallback(async () => {
    try {
      const [h, svc, tbl, eng, err, act] = await Promise.all([
        dashboardFetch("/v1/admin/monitor/health", token).then(r => r.ok ? r.json() : null),
        dashboardFetch("/v1/admin/monitor/services", token).then(r => r.ok ? r.json() : []),
        dashboardFetch("/v1/admin/monitor/tables", token).then(r => r.ok ? r.json() : []),
        dashboardFetch("/v1/admin/monitor/engine", token).then(r => r.ok ? r.json() : []),
        dashboardFetch("/v1/admin/monitor/errors?hours=24", token).then(r => r.ok ? r.json() : []),
        dashboardFetch("/v1/admin/activity?limit=50", token).then(r => r.ok ? r.json() : []),
      ]);
      setHealth(h);
      setServices(svc ?? []);
      setTables(tbl ?? []);
      setEngine(eng ?? []);
      setErrors(err ?? []);
      setActivity(act ?? []);
    } finally {
      setLoading(false);
    }
  }, [token]);

  useEffect(() => {
    fetchAll();
    const id = setInterval(fetchAll, 30_000);
    return () => clearInterval(id);
  }, [fetchAll]);

  const handleRestart = async (service: string) => {
    setRestarting(service);
    try {
      const res = await dashboardFetch(`/v1/admin/monitor/restart/${service}`, token, { method: "POST" });
      const msg = res.ok ? `${service.toUpperCase()} restarted` : `Failed (${res.status})`;
      setToast(msg);
      if (timerRef.current) clearTimeout(timerRef.current);
      timerRef.current = setTimeout(() => setToast(null), 3500);
    } finally {
      setRestarting(null);
    }
  };

  if (loading) return (
    <div style={{ display: "flex", alignItems: "center", justifyContent: "center", padding: 64 }}>
      <span style={{ fontFamily: S.fontMono, fontSize: 11, color: S.tertiary, letterSpacing: "0.1em" }}>LOADING…</span>
    </div>
  );

  const wiredCount = engine.filter(m => m.wired).length;

  return (
    <div style={{ maxWidth: 1400, margin: "0 auto", padding: "20px 24px", fontFamily: S.fontUI }}>
      {/* Toast */}
      {toast && (
        <div style={{ position: "fixed", bottom: 24, right: 24, zIndex: 9999, background: S.bgPanel, border: `1px solid ${S.cyan}`, padding: "10px 18px", fontFamily: S.fontMono, fontSize: 12, color: S.cyan, letterSpacing: "0.06em", boxShadow: "0 4px 20px rgba(0,0,0,0.3)" }}>
          {toast}
        </div>
      )}

      {/* KPI strip */}
      <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fit,minmax(140px,1fr))", gap: 12, marginBottom: 20 }}>
        {[
          { label: "UPTIME",      value: health?.uptime_human ?? "—",          color: S.pass },
          { label: "DB LATENCY",  value: health ? `${health.db_latency_ms}ms` : "—", color: health && health.db_latency_ms < 50 ? S.pass : S.amber },
          { label: "MEMORY RSS",  value: health?.memory ? `${health.memory.rss_mb}MB` : "—", color: S.secondary },
          { label: "TOTAL USERS", value: String(health?.total_users ?? "—"),   color: S.cyan },
          { label: "COMPANIES",   value: String(health?.total_companies ?? "—"), color: S.cyan },
          { label: "CALC RUNS",   value: String(health?.calc_runs ?? "—"),     color: S.amber },
          { label: "ENGINE MODS", value: `${wiredCount}/${engine.length} WIRED`, color: wiredCount === engine.length ? S.pass : S.amber },
        ].map(({ label, value, color }) => (
          <div key={label} style={{ background: S.bgPanel, border: `1px solid ${S.rim}`, padding: "12px 14px" }}>
            <div style={{ fontFamily: S.fontMono, fontSize: 10, color: S.tertiary, letterSpacing: "0.09em", marginBottom: 6 }}>{label}</div>
            <div style={{ fontFamily: S.fontMono, fontSize: 16, fontWeight: 700, color, lineHeight: 1.2 }}>{value}</div>
          </div>
        ))}
      </div>

      {/* Services */}
      <SectionCard title="SERVICE STATUS">
        <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fit,minmax(200px,1fr))", gap: 12 }}>
          {services.map(svc => (
            <div key={svc.name} style={{ background: S.bgSub, border: `1px solid ${S.rim}`, borderLeft: `3px solid ${STATUS_COLOR[svc.status] ?? S.tertiary}`, padding: "12px 14px" }}>
              <div style={{ display: "flex", alignItems: "center", gap: 8, marginBottom: 6 }}>
                <span style={{ width: 8, height: 8, borderRadius: "50%", background: STATUS_COLOR[svc.status] ?? S.tertiary, flexShrink: 0, display: "inline-block" }} />
                <span style={{ fontFamily: S.fontMono, fontSize: 11, fontWeight: 700, color: S.primary, letterSpacing: "0.06em" }}>{svc.name.toUpperCase()}</span>
              </div>
              <div style={{ fontFamily: S.fontMono, fontSize: 11, color: S.tertiary, marginBottom: 10 }}>{svc.uptime_human}</div>
              {(svc.name === "cache" || svc.name === "scheduler") && (
                <button
                  onClick={() => handleRestart(svc.name)}
                  disabled={restarting === svc.name}
                  style={{ fontFamily: S.fontMono, fontSize: 10, fontWeight: 700, letterSpacing: "0.07em", padding: "3px 10px", background: "transparent", border: `1px solid ${S.rim}`, color: S.secondary, cursor: "pointer" }}
                >
                  {restarting === svc.name ? "…" : svc.name === "cache" ? "CLEAR CACHE" : "RESTART"}
                </button>
              )}
            </div>
          ))}
        </div>
      </SectionCard>

      {/* DB Tables */}
      <SectionCard title={`DATABASE TABLES — ${tables.length} MONITORED`}>
        <div style={{ maxHeight: 280, overflowY: "auto" }}>

import { test, expect } from "@playwright/test";

const BASE = process.env.PLAYWRIGHT_BASE_URL ?? "http://localhost:3000";
const SU_EMAIL = process.env.TEST_SUPERUSER_EMAIL ?? "demo";
const SU_PASS  = process.env.TEST_SUPERUSER_PASS  ?? "demo";

async function loginAs(page: import("@playwright/test").Page, email: string, password: string) {
  await page.goto(`${BASE}/auth/login`);
  await page.fill('input[name="username"], input[type="text"]', email);
  await page.fill('input[name="password"], input[type="password"]', password);

  await page.waitForURL(/\/(dashboard|admin|position-desk)/, { timeout: 15_000 });

  });

  test("Operations tab — refresh does not crash", async ({ page }) => {

    await page.click('button:has-text("DEVOPS")');

    await expect(page.locator("text=SERVICE STATUS")).toBeVisible({ timeout: 15_000 });
  });

  test("Users tab — table renders", async ({ page }) => {
    await page.click('button:has-text("USERS")');
